Event Processing

Webhook Worker

Receive, validate, and process incoming webhooks. Events are stored in Postgres with full trace visibility.

Deploy in one command

$ flux create my-webhook --template webhook-worker
$ cd my-webhook
$ flux secrets set WEBHOOK_SECRET "your-signing-secret"
$ flux init --project <PROJECT_ID>
$ flux deploy

Database schema

CREATE TABLE IF NOT EXISTS events (
  id           UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  event_type   TEXT NOT NULL,
  source       TEXT,
  payload      JSONB,
  processed    BOOLEAN DEFAULT FALSE,
  received_at  TIMESTAMPTZ DEFAULT now()
);

Function

handle_event — POST /handle_event

// functions/handle_event/index.ts
import { defineFunction, type FluxContext } from "@fluxbase/functions";

interface WebhookPayload {
  event: string;
  source?: string;
  payload?: Record<string, unknown>;
}

async function verifySignature(
  signature: string,
  body: string,
  secret: string
): Promise<boolean> {
  // Replace with your signing mechanism (HMAC-SHA256, etc.)
  return signature.startsWith("sha256=");
}

export default defineFunction({
  name: "handle_event",
  handler: async ({ input, ctx }: { input: WebhookPayload; ctx: FluxContext }) => {
    // 1. Verify signature
    const secret = ctx.env.WEBHOOK_SECRET;
    const sig    = ctx.env.X_WEBHOOK_SIGNATURE ?? "";
    if (!(await verifySignature(sig, JSON.stringify(input), secret))) {
      throw new Error("Invalid webhook signature");
    }

    // 2. Log event
    ctx.log("received webhook", { event: input.event, source: input.source });

    // 3. Store in database
    const [record] = await ctx.db.query({
      table: "events", operation: "insert",
      data: {
        event_type: input.event,
        source:     input.source ?? "unknown",
        payload:    JSON.stringify(input.payload ?? {}),
      },
    });

    // 4. Route to handler based on event type
    switch (input.event) {
      case "payment.completed":
        // handle payment
        break;
      case "user.created":
        // handle user creation
        break;
    }

    return { received: true, id: record.id };
  },
});

Usage

# Register as webhook endpoint in Stripe, GitHub, etc.
https://gw.fluxbase.co/handle_event

# Test locally
$ flux invoke handle_event --payload \
    '{"event":"payment.completed","source":"stripe","payload":{"amount":4900}}'
{"received":true,"id":"49a3..."}

Query stored events

$ curl https://gw.fluxbase.co/list_events \
    -H "X-API-Key: $API_KEY" \
    -d '{"event_type":"payment.completed","limit":10}'
Async mode
Set "async": true in flux.json to process webhooks asynchronously. The gateway returns 202 Accepted immediately and queues the event for processing.
