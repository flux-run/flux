Todo API

Full CRUD with Postgres — create, list, update, and delete tasks using 4 serverless functions.

Deploy in one command

$ flux create my-todo-api --template todo-api
$ cd my-todo-api
$ flux init --project <PROJECT_ID>
$ flux deploy

Database schema

CREATE TABLE IF NOT EXISTS todos (
  id         UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  title      TEXT NOT NULL,
  done       BOOLEAN DEFAULT FALSE,
  user_id    TEXT,
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);

Functions

create_todo — POST /create_todo

// functions/create_todo/index.ts
import { defineFunction } from "@fluxbase/functions";

export default defineFunction({
  name: "create_todo",
  handler: async ({ input, ctx }) => {
    const [todo] = await ctx.db.query({
      table: "todos", operation: "insert",
      data: { title: input.title, done: false },
    });
    return todo;
  },
});

list_todos — GET /list_todos

export default defineFunction({
  name: "list_todos",
  handler: async ({ input, ctx }) => {
    const todos = await ctx.db.query({
      table: "todos", operation: "select",
      filters: input.done !== undefined
        ? [{ column: "done", op: "eq", value: input.done }]
        : [],
      limit: input.limit ?? 50,
    });
    return { todos };
  },
});

update_todo — POST /update_todo

export default defineFunction({
  name: "update_todo",
  handler: async ({ input, ctx }) => {
    await ctx.db.query({
      table: "todos", operation: "update",
      filters: [{ column: "id", op: "eq", value: input.id }],
      data: { done: input.done, updated_at: new Date().toISOString() },
    });
    return { ok: true };
  },
});

delete_todo — POST /delete_todo

export default defineFunction({
  name: "delete_todo",
  handler: async ({ input, ctx }) => {
    await ctx.db.query({
      table: "todos", operation: "delete",
      filters: [{ column: "id", op: "eq", value: input.id }],
    });
    return { ok: true };
  },
});

Usage examples

# Create a todo
$ curl -X POST https://gw.fluxbase.co/create_todo \
    -H "X-API-Key: $API_KEY" \
    -H "Content-Type: application/json" \
    -d '{"title":"Launch on Fluxbase"}'
{"id":"c7f2...","title":"Launch on Fluxbase","done":false}

# List todos
$ curl https://gw.fluxbase.co/list_todos \
    -H "X-API-Key: $API_KEY"
{"todos":[{"id":"c7f2...","title":"Launch on Fluxbase","done":false}]}

# Mark done
$ curl -X POST https://gw.fluxbase.co/update_todo \
    -H "X-API-Key: $API_KEY" \
    -H "Content-Type: application/json" \
    -d '{"id":"c7f2...","done":true}'

# Delete
$ curl -X POST https://gw.fluxbase.co/delete_todo \
    -H "X-API-Key: $API_KEY" \
    -H "Content-Type: application/json" \
    -d '{"id":"c7f2..."}'
